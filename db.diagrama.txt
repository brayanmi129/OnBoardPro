Table User {
  id string [pk]
  firstname string
  lastname string
  phonenumber number
  email string
  level number
  password string
  role string
  status string
  xp number
}

Table Group {
  id string [pk]
  name string
}

Table Course {
  id string [pk]
  name string
  instructor_id string [ref: > User.id]
  status string
}

Table Activity {
  id string [pk]
  course_id string [ref: > Course.id]
  title string
  description string
  attachment url
  deliverable boolean
  type string
  xp number
}
